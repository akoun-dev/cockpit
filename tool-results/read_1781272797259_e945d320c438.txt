   138тЖТ   138тЖТfunction formatValue(value: number | null | undefined, unit: string): string {
   139тЖТ   139тЖТ  if (value == null) return 'тАФ';
   140тЖТ   140тЖТ  if (unit === 'nb' || unit === 'km' || unit === 'h' || unit === 'jours')
   141тЖТ   141тЖТ    return Math.round(value).toLocaleString('fr-FR');
   142тЖТ   142тЖТ  if (unit === 'Mds FCFA') return `${value.toFixed(2)} Mds`;
   143тЖТ   143тЖТ  if (unit === 'ratio') return value.toFixed(2);
   144тЖТ   144тЖТ  return `${value.toFixed(1)}${unit === '%' ? '%' : ''}`;
   145тЖТ   145тЖТ}

   156тЖТ   156тЖТfunction computeStatus(
   157тЖТ   157тЖТ  value: number | null,
   158тЖТ   158тЖТ  target: number | null,
   159тЖТ   159тЖТ  unit: string
   160тЖТ   160тЖТ): StatusType {
   161тЖТ   161тЖТ  if (value == null || target == null || target === 0) return 'non_atteint';
   162тЖТ   162тЖТ  const lower = isLowerBetter(unit);
   163тЖТ   163тЖТ  if (lower) {
   164тЖТ   164тЖТ    if (value <= target) return 'atteint';
   165тЖТ   165тЖТ    if (value <= target * 1.5) return 'partiel';
   166тЖТ   166тЖТ    return 'non_atteint';
   167тЖТ   167тЖТ  } else {
   168тЖТ   168тЖТ    if (value >= target) return 'atteint';
   169тЖТ   169тЖТ    if (value >= target * 0.8) return 'partiel';
   170тЖТ   170тЖТ    return 'non_atteint';
   171тЖТ   171тЖТ  }
   172тЖТ   172тЖТ}
   173тЖТ   173тЖТ
   174тЖТ   174тЖТfunction computeEcart(
   175тЖТ   175тЖТ  value: number | null,
   176тЖТ   176тЖТ  target: number | null,
   177тЖТ   177тЖТ  unit: string

   181тЖТ   181тЖТ  const lower = isLowerBetter(unit);
   182тЖТ   182тЖТ  const positive = lower ? ecart <= 0 : ecart >= 0;
   183тЖТ   183тЖТ  const sign = ecart >= 0 ? '+' : '';
   184тЖТ   184тЖТ  let suffix = '';
   185тЖТ   185тЖТ  if (unit === '%') suffix = ' pts';
   186тЖТ   186тЖТ  else if (unit === 'Mds FCFA') suffix = ' Mds';
   187тЖТ   187тЖТ  else if (unit === 'jours') suffix = ' j';
   188тЖТ   188тЖТ  else if (unit === 'h') suffix = ' h';
   189тЖТ   189тЖТ  else if (unit === 'km') suffix = ' km';
   190тЖТ   190тЖТ  else if (unit === 'ratio') suffix = '';
   191тЖТ   191тЖТ  else if (unit === 'nb') suffix = '';
   192тЖТ   192тЖТ  else suffix = unit ? ` ${unit}` : '';
   193тЖТ   193тЖТ  return {
   194тЖТ   194тЖТ    text: `${sign}${ecart.toFixed(unit === 'ratio' || unit === 'Mds FCFA' ? 2 : 1)}${suffix}`,

   602тЖТ   602тЖТ

   633тЖТ   633тЖТ    }

   664тЖТ   664тЖТ    const params = new URLSearchParams({ domain, year: String(filters.year || 2025) });
   665тЖТ   665тЖТ    if (filters.quarter) params.set('quarter', String(filters.quarter));

   682тЖТ   682тЖТ        }

   685тЖТ   685тЖТ    return () => {
   686тЖТ   686тЖТ      cancelled = true;
   687тЖТ   687тЖТ    };

   702тЖТ   702тЖТ        <div className="flex items-center gap-2 min-w-0">
   703тЖТ   703тЖТ          <TableProperties className="size-4 text-fun-blue shrink-0" />

   714тЖТ   714тЖТ      {subDomainKeys.length > 0 && (
